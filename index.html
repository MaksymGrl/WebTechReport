Система онлайн бронювання:

Вибір послуг: Дозволяє клієнтам вибирати зі списку доступних послуг (наприклад, стрижка, підстригання бороди тощо).

Вибір перукаря: Надає можливість вибору перукаря, якщо клієнт має відповідні побажання.

Вибір часових інтервалів: Відображає доступні часові проміжки в режимі реального часу.

Управління обліковими записами клієнтів:

Створення профілю: Дозволяє клієнтам створювати та керувати своїми профілями, зберігаючи особисту інформацію та вподобання.

Історія бронювання: Дозволяє клієнтам переглядати минулі зустрічі та майбутні бронювання.

Перенесення та скасування: Клієнт має можливість змінювати або скасовувати зустрічі через свій обліковий запис.

Відгуки після надання послуг: Після кожної зустрічі клієнт може оцінити сервіс і залишити відгук щоб покращити якість обслуговування.

Платіжна інтеграція:

Виставлення рахунків: Створює та надсилає цифрові рахунки за надані послуги.

Рекламні функції:

Сторінка спеціальних пропозицій: Розділ для поточних акцій та спеціальних пропозицій.

Подарункові картки та ваучери: Дозволяє купувати подарункові картки через сайт.

В рамках першого завдання цієї лабораторної роботи був розроблен макет головної сторінки веб-застосунку. Була використана блокова гумова верстка. Цей тип верстки зберігає пропорції між блоками та реагує на зміну розміра та співвідношення viewport.

Завдання 2

В другому завданні були розроблени 4 макети відповідно до варіанту: табличний гумовий і фіксований, блоковий гумовий і фіксований. З цих макетів я отримав наступні висновки: Блочна верстка є більш простою у реалізації та підходить до більшості макетів, але найкращим варіантом может бути використання обох типів. Фіксована верстка є простішою у реалізації, порівняючи з гумовою, та надає можливість надати кращий дизайн на тергетному екрані. Але є великий недолік - відсутність адаптивності, що робить гумову або змішану верстку більш бажаною.

Завдання 3

Третє завдання присвячене використанню flexbox і верстці макета сайту за зразком. У результаті отримано макет, який відповідає вимогам, зразку і плавно реагує на зміну розміру екрана, що є плюсом цього типу верстки. З мінусів це складність у розробці та налаштуванні.

                function swapMaxAndFirst(arr) {
                    let maxIndex = 0;
                    for (let i = 1; i < arr.length; i++) {
                        if (arr[i] > arr[maxIndex]) {
                            maxIndex = i;
                        }
                    }
                    [arr[0], arr[maxIndex]] = [arr[maxIndex], arr[0]];
                }

                function bubbleSort(arr) {
                    let len = arr.length;
                    for (let i = 0; i < len; i++) {
                        for (let j = 0; j < len - 1; j++) {
                            if (arr[j] > arr[j + 1]) {
                                [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]];
                            }
                        }
                    }

Напишіть функцію checkBrackets(str) яка приймає рядок жс коду (someFn) і перевіряє правильність закриття дужок () {} []

Якщо рядок містить коректний код функція повертає true.

В іншому випадку повертає false


                function checkBrackets(str) {
                    const stack = [];
                    const brackets = {
                        '(': ')',
                        '{': '}',
                        '[': ']'
                    };
            
                    for (let char of str) {
                        if (brackets[char]) {
                            // Якщо символ - це відкрита дужка, додаємо її до стека
                            stack.push(char);
                        } else if (Object.values(brackets).includes(char)) {
                            // Якщо символ - це закрита дужка, перевіряємо чи відповідає вона останній відкритій дужці
                            if (brackets[stack.pop()] !== char) {
                                return false;
                            }
                        }
                    }
            
                    // Якщо стек порожній, то всі дужки правильно закриті
                    return stack.length === 0;
                }
            
                // Приклади використання
                console.log(checkBrackets("function someFn() { return { key: 'value' }; }")); // true
                console.log(checkBrackets("function someFn() { return { key: 'value' }"));   // false
                console.log(checkBrackets("if (a > b) { console.log(a); } else { console.log(b); }")); // true
                console.log(checkBrackets("[{()}]")); // true
                console.log(checkBrackets("[{(])}")); // false

                function App() {
                const inputRef = useRef(null);
                const resultRef = useRef(null);
                const [result, setResult] = useState(0);

                function plus(e) {
                    e.preventDefault();
                    setResult((result) => result + Number(inputRef.current.value));
                }

                function minus(e) {
                    e.preventDefault();
                    setResult((result) => result - Number(inputRef.current.value));
                }

                function times(e) {
                    e.preventDefault();
                    setResult((result) => result * Number(inputRef.current.value));
                }

                function divide(e) {
                    e.preventDefault();
                    setResult((result) => result / Number(inputRef.current.value));
                }

                function resetInput(e) {
                    e.preventDefault();
                    inputRef.current.value = '';
                }

                function resetResult(e) {
                    e.preventDefault();
                    setResult(0);
                }

                return (
                    <div className="App">
                    <div>
                    <h1>Simplest Working Calculator</h1>
                    </div>
                    <form>
                    <p ref={resultRef}>
                        {result}
                        </p>
                        <input
                        pattern="[0-9]"
                        ref={inputRef}
                        type="number"
                        placeholder="Type a number"
                        />
                        <button onClick={plus}>Add</button>
                        <button onClick={minus}>Subtract</button>
                        <button onClick={times}>Multiply</button>
                        <button onClick={divide}>Divide</button>
                        <button onClick={resetInput}>Reset Input</button>
                        <button onClick={resetResult}>Reset Result </button>
                    </form>
                    </div>
                );
                }

                export default App;

                const express = require('express');
                const { ObjectId } = require('mongodb');
                const client = require('./db');
                const jwt = require('jsonwebtoken');

                const app = express();
                const port = 3000;

                app.use(express.json());

                const dbName = 'mydatabase';
                const collectionName = 'items';

                // Middleware for error handling
                app.use((err, req, res, next) => {
                console.error(err.stack);
                res.status(500).json({ error: err.message });
                });

                // Middleware for authentication
                const authenticateToken = (req, res, next) => {
                const token = req.header('Authorization')?.split(' ')[1];
                if (!token) return res.sendStatus(401);

                jwt.verify(token, 'your_jwt_secret', (err, user) => {
                    if (err) return res.sendStatus(403);
                    req.user = user;
                    next();
                });
                };

                // Middleware for role-based authorization
                const authorizeRole = (role) => {
                return (req, res, next) => {
                    if (req.user.role !== role) return res.sendStatus(403);
                    next();
                };
                };

                // CRUD routes
                app.post('/items', authenticateToken, async (req, res, next) => {
                const item = req.body;
                try {
                    const db = client.db(dbName);
                    const result = await db.collection(collectionName).insertOne(item);
                    res.status(201).json(result.ops[0]);
                } catch (err) {
                    next(err);
                }
                });

                app.get('/items', authenticateToken, async (req, res, next) => {
                try {
                    const db = client.db(dbName);
                    const items = await db.collection(collectionName).find().toArray();
                    res.status(200).json(items);
                } catch (err) {
                    next(err);
                }
                });

                app.put('/items/:id', authenticateToken, async (req, res, next) => {
                const { id } = req.params;
                const update = req.body;
                try {
                    const db = client.db(dbName);
                    const result = await db.collection(collectionName).findOneAndUpdate(
                    { _id: new ObjectId(id) },
                    { $set: update },
                    { returnOriginal: false }
                    );
                    res.status(200).json(result.value);
                } catch (err) {
                    next(err);
                }
                });

                app.delete('/items/:id', authenticateToken, async (req, res, next) => {
                const { id } = req.params;
                try {
                    const db = client.db(dbName);
                    await db.collection(collectionName).deleteOne({ _id: new ObjectId(id) });
                    res.status(204).send();
                } catch (err) {
                    next(err);
                }
                });

                // Admin-only route
                app.post('/admin/items', authenticateToken, authorizeRole('admin'), async (req, res, next) => {
                const item = req.body;
                try {
                    const db = client.db(dbName);
                    const result = await db.collection(collectionName).insertOne(item);
                    res.status(201).json(result.ops[0]);
                } catch (err) {
                    next(err);
                }
                });

                app.listen(port, () => {
                console.log(`Server is running on http://localhost:${port}`);
                });
